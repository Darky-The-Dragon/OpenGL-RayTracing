6// Compute the point light position in world space,
7// optionally orbiting around the base position.
8glm::vec3 computePointLightWorldPos(const RenderParams &params) {
9 glm::vec3 base(params.pointLightPos[0],
10 params.pointLightPos[1],
11 params.pointLightPos[2]);
12
13 if (!params.pointLightOrbitEnabled || params.pointLightOrbitRadius <= 0.0f) {
14 return base;
15 }
16
17 float yawRad = glm::radians(params.pointLightYaw);
18 float pitchRad = glm::radians(params.pointLightPitch);
19
20 float cy = cosf(yawRad);
21 float sy = sinf(yawRad);
22 float cp = cosf(pitchRad);
23 float sp = sinf(pitchRad);
24
25 glm::vec3 dir;
26 dir.x = cp * sy;
27 dir.y = sp;
28 dir.z = cp * cy;
29
30 return base + dir * params.pointLightOrbitRadius;
31}

33// Build a normalized direction vector from yaw/pitch angles (in degrees).
34// Used to drive sun and sky directions.
35static glm::vec3 dirFromYawPitch(float yawDeg, float pitchDeg) {
36 float yaw = glm::radians(yawDeg);
37 float pitch = glm::radians(pitchDeg);
38
39 float cp = std::cos(pitch);
40 float sp = std::sin(pitch);
41 float cy = std::cos(yaw);
42 float sy = std::sin(yaw);
43
44 glm::vec3 d(cp * cy, sp, cp * sy);
45
46 // Fallback if direction degenerates
47 if (glm::dot(d, d) < 1e-6f) {
48 return {0.0f, -1.0f, 0.0f};
49 }
50 return glm::normalize(d);
51}
