4/*
5 rt_taa.glsl – Temporal Anti-Aliasing (TAA) with History Variance Tracking
6
7 This module implements a simple but robust TAA resolve used by the ray
8 tracing pipeline. It operates on:
9
10 - curr : current frame color (already averaged over SPP)
11 - motionOut : NDC motion vector (currNDC - prevNDC)
12 - prevAccum : history texture (RGB = accumulated color, A = second moment of luma M2)
13 - frameIndex: accumulation frame counter from rt::Accum
14
15 Features:
16 - Switchable TAA (uEnableTAA): when disabled, the pass simply forwards
17 the current color but still updates M2 so the SVGF filter can work.
18 - Separate handling for:
19 * Static pixels → history accumulation with configurable weights.
20 * Moving pixels → reprojection using motion vectors.
21 - Basic color-based confidence:
22 * Large luminance differences reduce or kill history to avoid smearing.
23 - History clamping:
24 * History color is clamped to a small box around the current color to
25 suppress outliers and fireflies before blending.
26
27 Tuning:
28 - uTaaStillThresh, uTaaHardMovingThresh: control when a pixel is considered
29 static vs moving, and when history is fully discarded.
30 - uTaaHistoryMinWeight / Avg / Max: control how quickly history converges.
31 - uTaaHistoryBoxSize: controls the clamp region size around the current color.
32*/
33
34/**
35 * @brief TAA resolve, combining current frame with reprojected history.
36 *
37 * @param curr Current frame linear color (already averaged over SPP for this frame).
38 * @param uvCurr UV coordinates of the current pixel.
39 * @param motionOut NDC motion vector (currNDC - prevNDC) from the ray pass.
40 * @param prevAccum History texture: RGB = color, A = second moment of luma (M2).
41 * @param frameIndex Accumulation frame index (0,1,2,...) from rt::Accum.
42 *
43 * @return vec4:
44 * - rgb = TAA-resolved linear color
45 * - a = updated second moment of luma (M2) for variance estimation
46 */
47vec4 resolveTAA(vec3 curr, vec2 uvCurr, vec2 motionOut, sampler2D prevAccum, int frameIndex)
48{
49 // Luma coefficients (approx. Rec.709)
50 const vec3 YCOEFF = vec3(0.299, 0.587, 0.114);
51
52 float lCurr = dot(curr, YCOEFF);
53 float lCurr2 = lCurr * lCurr;
54
55 // ---------------------------------------------
56 // TAA disabled → return raw color + M2
57 // ---------------------------------------------
58 if (uEnableTAA == 0) {
59 // Still store M2 so SVGF can compute variance correctly.
60 return vec4(curr, lCurr2);
61 }
62
63 // ---------------------------------------------
64 // First frame: no valid history yet
65 // ---------------------------------------------
66 if (frameIndex == 0) {
67 return vec4(curr, lCurr2);
68 }
69
70 // Motion magnitude in NDC
71 float motMag = length(motionOut);
72
73 // CPU-driven params
74 float MIN_W_HIST = uTaaHistoryMinWeight;
75 float AVG_W_HIST = uTaaHistoryAvgWeight;
76 float MAX_W_HIST = uTaaHistoryMaxWeight;
77 float BOX_SIZE = uTaaHistoryBoxSize;
78
79 // ---------------------------------------------
80 // CASE 1: camera/pixel effectively still
81 //
82 // When motion is below uTaaStillThresh we:
83 // - sample history at the same UV
84 // - blend using temporal weights that depend on frameIndex
85 // ---------------------------------------------
86 if (motMag < uTaaStillThresh) {
87 vec4 prevRGBA = texture(prevAccum, uvCurr);
88 vec3 prevCol = prevRGBA.rgb;
89 float prevM2 = prevRGBA.a;
90
91 float wHist;
92 if (frameIndex < 8) {
93 wHist = MIN_W_HIST;
94 } else if (frameIndex < 32) {
95 wHist = AVG_W_HIST;
96 } else {
97 wHist = MAX_W_HIST;
98 }
99 float wCurr = 1.0 - wHist;
100
101 vec3 meanNew = prevCol * wHist + curr * wCurr;
102 float m2New = prevM2 * wHist + lCurr2 * wCurr;
103
104 return vec4(meanNew, m2New);
105 }
106
107 // ---------------------------------------------
108 // CASE 2: pixel is moving → motion-vector TAA
109 //
110 // - Reproject to previous frame using motionOut.
111 // - Sample history at uvPrev.
112 // - Apply motion- and color-based confidence to history.
113 // ---------------------------------------------
114
115 // Reproject into previous frame
116 vec2 uvPrev = uvCurr - motionOut * 0.5;
117
118 // Disocclusion → no reliable history
119 bool oob =
120 any(lessThan(uvPrev, vec2(0.0))) ||
121 any(greaterThan(uvPrev, vec2(1.0)));
122
123 if (oob) {
124 return vec4(curr, lCurr2);
125 }
126
127 // Fetch previous color and M2
128 vec4 prevRGBA = texture(prevAccum, uvPrev);
129 vec3 prevCol = prevRGBA.rgb;
130 float prevM2 = prevRGBA.a;
131
132 // ---------------------------------------------
133 // History weighting
134 //
135 // 1) Base history weight decreases as motion increases.
136 // 2) Luma-based confidence shrinks history when color changes a lot.
137 // 3) A hard kill-switch handles strong motion + color edges (sky/geometry).
138 // ---------------------------------------------
139
140 float wHist = 1.0 - smoothstep(0.02, uTaaHardMovingThresh, motMag);
141
142 // If motion is really large, just kill history completely
143 if (motMag > uTaaHardMovingThresh) {
144 wHist = 0.0;
145 }
146
147 // Color-based confidence using luma
148 float lPrev = dot(prevCol, YCOEFF);
149 float maxL = max(max(lCurr, lPrev), 1e-3);
150 float relDiff = abs(lCurr - lPrev) / maxL;
151
152 // Small relDiff → keep history, large relDiff → shrink history
153 float colorWeight = 1.0 - smoothstep(0.03, 0.25, relDiff);
154 wHist *= colorWeight;
155
156 // --- Extra kill-switch for strong color changes (sky/geometry edges) ---
157 bool bigColorChange = (motMag > 0.02) && (relDiff > 0.30); // sensitive enough to kill plane/sky streaks
158
159 if (bigColorChange) {
160 wHist = 0.0;
161 }
162
163 // Clamp so history never fully dominates (uses param instead of literal 0.90)
164 wHist = clamp(wHist, 0.0, MAX_W_HIST);
165 float wCurr = 1.0 - wHist;
166
167 // History clamping box
168 vec3 historyCol = prevCol;
169 vec3 lo = curr - vec3(BOX_SIZE);
170 vec3 hi = curr + vec3(BOX_SIZE);
171 historyCol = clamp(historyCol, lo, hi);
172
173 // Final TAA blended color
174 vec3 taaCol = wHist * historyCol + wCurr * curr;
175
176 // Temporal variance (M2) updated with the same weights
177 float m2New = wHist * prevM2 + wCurr * lCurr2;
178
179 return vec4(taaCol, m2New);
180}
