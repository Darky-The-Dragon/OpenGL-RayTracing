5/*
6 rt_scene_analytic.glsl – Analytic Test Scene & Sky
7
8 This module defines a small, hard-coded “analytic” scene used by the
9 ray tracer. All geometry is defined directly in GLSL:
10
11 - Infinite plane (floor).
12 - Left diffuse sphere (GUI-controlled material).
13 - Glass sphere.
14 - Mirror sphere.
15 - Small marker sphere placed at the point light position.
16
17 It provides:
18 - Basic plane and sphere intersection routines.
19 - A traceAnalyticCore() function that can optionally ignore:
20 * the glass sphere (for glass rays),
21 * the point-light marker sphere (for shadow rays to the bulb).
22 - Three public trace functions:
23 * traceAnalytic() → full scene
24 * traceAnalyticIgnoreGlass() → no glass sphere
25 * traceAnalyticIgnorePointLight()→ no point-light marker sphere
26 - A sky() function that samples either:
27 * an environment cubemap (if enabled), or
28 * a simple analytic gradient sky (fallback).
29
30 The material IDs (MAT_*) are defined in rt_materials.glsl and must
31 stay consistent with this setup.
32*/
33

58/**
59 * @brief Intersects a ray with an infinite plane.
60 *
61 * Plane equation: dot(n, x) + d = 0
62 *
63 * @param ro Ray origin.
64 * @param rd Ray direction (assumed normalized).
65 * @param n Plane normal (normalized).
66 * @param d Plane offset.
67 * @param h Output hit record (filled on success).
68 * @param matId Material ID to assign if hit.
69 * @return True if the ray hits the plane at t >= uEPS.
70 */
71bool intersectPlane(vec3 ro, vec3 rd, vec3 n, float d, out Hit h, int matId) {
72 float denom = dot(n, rd);
73 if (abs(denom) < 1e-6) return false;
74 float t = -(dot(n, ro) + d) / denom;
75 if (t < uEPS) return false;
76 h.t = t;
77 h.p = ro + rd * t;
78 h.n = n;
79 h.mat = matId;
80 return true;
81}

83/**
84 * @brief Intersects a ray with a sphere.
85 *
86 * Sphere equation: |x - c|^2 = r^2
87 *
88 * @param ro Ray origin.
89 * @param rd Ray direction (assumed normalized).
90 * @param c Sphere center.
91 * @param r Sphere radius.
92 * @param h Output hit record (filled on success).
93 * @param matId Material ID to assign if hit.
94 * @return True if the ray hits the sphere at t >= uEPS.
95 */
96bool intersectSphere(vec3 ro, vec3 rd, vec3 c, float r, out Hit h, int matId) {
97 vec3 oc = ro - c;
98 float b = dot(oc, rd);
99 float c2 = dot(oc, oc) - r * r;
100 float disc = b * b - c2;
101 if (disc < 0.0) return false;
102 float s = sqrt(disc);
103 float t = -b - s;
104 if (t < uEPS) t = -b + s;
105 if (t < uEPS) return false;
106 h.t = t;
107 h.p = ro + rd * t;
108 h.n = normalize(h.p - c);
109 h.mat = matId;
110 return true;
111}

113/**
114 * @brief Core analytic scene tracer with optional inclusion flags.
115 *
116 * This function traces against:
117 * - Floor (always)
118 * - Left diffuse sphere (always)
119 * - Glass sphere (controlled by includeGlass)
120 * - Mirror sphere (always)
121 * - Point-light marker sphere (controlled by includePointLightSphere & uPointLightEnabled)
122 *
123 * It returns the closest hit in @p hit if any intersection is found.
124 *
125 * @param ro Ray origin.
126 * @param rd Ray direction.
127 * @param includeGlass Include the glass sphere in intersection tests.
128 * @param includePointLightSphere Include the point-light marker sphere.
129 * @param hit Output closest hit (valid only if function returns true).
130 * @return True if any object was hit; false otherwise.
131 */
132bool traceAnalyticCore(vec3 ro, vec3 rd, bool includeGlass, bool includePointLightSphere, out Hit hit) {
133 hit.t = uINF;
134 Hit h;
135
136 // Floor
137 if (intersectPlane(ro, rd, kFloorNormal, kFloorD, h, MAT_FLOOR) && h.t < hit.t) {
138 hit = h;
139 }
140
141 // Left albedo sphere
142 if (intersectSphere(ro, rd, kSphereLeftCenter, kSphereLeftRadius, h, MAT_ALBEDO_SPHERE) && h.t < hit.t) {
143 hit = h;
144 }
145
146 // Glass sphere (optionally ignored)
147 if (includeGlass) {
148 if (intersectSphere(ro, rd, kGlassCenter, kGlassRadius, h, MAT_GLASS_SPHERE) && h.t < hit.t) {
149 hit = h;
150 }
151 }
152
153 // Mirror sphere
154 if (intersectSphere(ro, rd, kMirrorCenter, kMirrorRadius, h, MAT_MIRROR_SPHERE) && h.t < hit.t) {
155 hit = h;
156 }
157
158 // Point-light marker sphere (optional)
159 if (includePointLightSphere && uPointLightEnabled == 1) {
160 // Center is the actual point light position
161 if (intersectSphere(ro, rd, uPointLightPos, kPointLightRadius, h, MAT_POINTLIGHT_SPHERE) && h.t < hit.t) {
162 hit = h;
163 }
164 }
165
166 return hit.t < uINF;
167}

169/**
170 * @brief Full analytic scene ray trace (floor + all spheres + marker).
171 *
172 * This is the general-purpose analytic tracing function used for primary
173 * rays and most shading paths.
174 */
175bool traceAnalytic(vec3 ro, vec3 rd, out Hit hit) {
176 return traceAnalyticCore(ro, rd, true, true, hit);
177}

179/**
180 * @brief Analytic scene trace that ignores the glass sphere.
181 *
182 * Useful for refraction/glass paths where we want to see what lies
183 * behind the glass without intersecting the glass surface itself.
184 */
185bool traceAnalyticIgnoreGlass(vec3 ro, vec3 rd, out Hit hit) {
186 return traceAnalyticCore(ro, rd, false, true, hit);
187}

189/**
190 * @brief Analytic scene trace that ignores the point-light marker sphere.
191 *
192 * Used for shadow rays toward the point light so that the emissive
193 * marker sphere does not shadow its own light.
194 */
195bool traceAnalyticIgnorePointLight(vec3 ro, vec3 rd, out Hit hit) {
196 return traceAnalyticCore(ro, rd, true, false, hit);
197}

201/**
202 * @brief Environment lighting lookup.
203 *
204 * If uUseEnvMap == 1, samples a cubemap environment and scales it.
205 * Otherwise, returns a simple analytic gradient sky:
206 * - lighter near the horizon, darker towards the zenith.
207 *
208 * @param dir Normalized direction vector.
209 * @return Environment radiance in that direction.
210 */
211vec3 sky(vec3 dir) {
212 // If an environment cubemap is enabled, use it; otherwise fall back to analytic sky.
213 if (uUseEnvMap == 1) {
214 vec3 env = texture(uEnvMap, dir).rgb;
215 return env * uEnvIntensity;
216 }
217
218 float t = clamp(0.5 * (dir.y + 1.0), 0.0, 1.0);
219 vec3 col = mix(vec3(0.6, 0.7, 0.9) * 0.3,
220 vec3(0.1, 0.15, 0.3) * 0.3,
221 1.0 - t);
222 return col;
223}
