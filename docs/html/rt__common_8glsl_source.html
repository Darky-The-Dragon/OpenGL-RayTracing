48/**
49 * @brief Simple 2D integer hash (32-bit) for RNG.
50 *
51 * Based on a small LCG-style mixing sequence. Used as a building block
52 * for float-valued random numbers.
53 *
54 * @param v Input 2D unsigned integer vector.
55 * @return Pseudo-random 32-bit unsigned int.
56 */
57uint hash2(uvec2 v) {
58 v = v * 1664525u + 1013904223u;
59 v.x ^= v.y >> 16;
60 v.y ^= v.x << 5;
61 v = v * 1664525u + 1013904223u;
62 return v.x ^ v.y;
63}

65/**
66 * @brief Float RNG in [0,1) based on integer hashing.
67 *
68 * Combines pixel coordinates and frame index to produce a reproducible
69 * but decorrelated random value.
70 *
71 * @param p 2D position (typically pixel coords).
72 * @param frame Frame index used to vary the pattern over time.
73 * @return Pseudo-random float in [0,1).
74 */
75float rand(vec2 p, int frame) {
76 return float(hash2(uvec2(p) ^ uvec2(frame, frame * 1663))) / 4294967296.0;
77}

96/**
97 * @brief Computes the i-th sample of a 1D Halton sequence base b.
98 *
99 * Used to generate quasi-random samples with better stratification than
100 * pure random numbers, especially for TAA jitter and sampling patterns.
101 *
102 * @param i Sample index.
103 * @param b Base of the Halton sequence (e.g., 2 or 3).
104 * @return Halton value in [0,1).
105 */
106float halton(int i, int b) {
107 float f = 1.0;
108 float r = 0.0;
109 int n = i;
110 while (n > 0) {
111 f /= float(b);
112 r += f * float(n % b);
113 n /= b;
114 }
115 return r;
116}
117
118/**
119 * @brief 2D low-discrepancy sequence using Halton bases 2 and 3.
120 *
121 * ld2(i) = (Halton(i+1, 2), Halton(i+1, 3)).
122 * Useful for jitter patterns and sampling across frames.
123 *
124 * @param i Sample index.
125 * @return 2D low-discrepancy sample in [0,1)^2.
126 */
127vec2 ld2(int i) {
128 return vec2(halton(i + 1, 2), halton(i + 1, 3));
129}

135/**
136 * @brief Concentric mapping from [0,1]^2 to a unit disk.
137 *
138 * Converts a uniform 2D random variable into a uniform disk sample using
139 * the "concentric mapping" technique (better distribution than naive polar).
140 *
141 * @param u 2D uniform sample in [0,1]^2.
142 * @return 2D sample on a unit disk (x,y).
143 */
144vec2 concentricSample(vec2 u) {
145 float a = 2.0 * u.x - 1.0;
146 float b = 2.0 * u.y - 1.0;
147 float r, phi;
148 if (a == 0.0 && b == 0.0) {
149 r = 0.0;
150 phi = 0.0;
151 } else if (abs(a) > abs(b)) {
152 r = a;
153 phi = (uPI / 4.0) * (b / a);
154 } else {
155 r = b;
156 phi = (uPI / 2.0) - (uPI / 4.0) * (a / b);
157 }
158 return r * vec2(cos(phi), sin(phi));
159}

165/**
166 * @brief Projects a world-space position into NDC.
167 *
168 * Used to compute motion vectors (currentNDC - prevNDC) by applying
169 * the current and previous view-projection matrices.
170 *
171 * @param p World-space position.
172 * @param VP View-projection matrix.
173 * @return NDC coordinates in [-1,1]^2.
174 */
175vec2 ndcFromWorld(vec3 p, mat4 VP) {
176 vec4 clip = VP * vec4(p, 1.0);
177 vec3 ndc = clip.xyz / max(clip.w, 1e-6);
178 return ndc.xy; // [-1,1]
179}
