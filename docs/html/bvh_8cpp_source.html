35// Recursive median-split BVH builder.
36// - nodes: output BVH nodes
37// - tris: CPU triangle array
38// - refs: references into tris, sorted by centroid along split axis
39// - [begin, end): range of refs this node owns
40// - leafMax: max triangles per leaf
41static int build_recursive(std::vector<BVHNode> &nodes,
42 const std::vector<CPU_Triangle> &tris,
43 std::vector<BuildRef> &refs,
44 const int begin,
45 const int end,
46 const int leafMax = 8) {
47 // Compute bounding box over this node's triangle range.
48 glm::vec3 bMin(1e30f), bMax(-1e30f);
49 for (int i = begin; i < end; ++i) {
50 const auto &T = tris[refs[i].triIndex];
51 bMin = glm::min(bMin, tri_min(T));
52 bMax = glm::max(bMax, tri_max(T));
53 }
54
55 const int count = end - begin;
56 const int myIndex = static_cast<int>(nodes.size());
57 nodes.push_back({});
58 nodes[myIndex].bMin = bMin;
59 nodes[myIndex].bMax = bMax;
60
61 // Leaf: store range and triangle count, no children.
62 if (count <= leafMax) {
63 nodes[myIndex].left = -1;
64 nodes[myIndex].right = -1;
65 nodes[myIndex].first = begin;
66 nodes[myIndex].count = count;
67 return myIndex;
68 }
69
70 // Choose split axis by largest extent.
71 const glm::vec3 e = bMax - bMin;
72 int axis = (e.x > e.y) ? ((e.x > e.z) ? 0 : 2) : ((e.y > e.z) ? 1 : 2);
73
74 // Median split along chosen axis (by centroid).
75 const int mid = (begin + end) / 2;
76 std::nth_element(refs.begin() + begin,
77 refs.begin() + mid,
78 refs.begin() + end,
79 [axis](const BuildRef &a, const BuildRef &b) {
80 return a.c[axis] < b.c[axis];
81 });
82
83 const int leftIdx = build_recursive(nodes, tris, refs, begin, mid, leafMax);
84 const int rightIdx = build_recursive(nodes, tris, refs, mid, end, leafMax);
85
86 nodes[myIndex].left = leftIdx;
87 nodes[myIndex].right = rightIdx;
88 nodes[myIndex].first = -1;
89 nodes[myIndex].count = 0;
90 return myIndex;
91}

93// Entry point: build BVH over tris, then remap them for cache-friendly leaves.
94std::vector<BVHNode> build_bvh(std::vector<CPU_Triangle> &tris) {
95 std::vector<BVHNode> nodes;
96 if (tris.empty()) return nodes;
97
98 // Build initial refs with centroids for splitting.
99 std::vector<BuildRef> refs(tris.size());
100 for (size_t i = 0; i < tris.size(); ++i) {
101 refs[i].triIndex = static_cast<int>(i);
102 refs[i].c = tri_centroid(tris[i]);
103 }
104
105 nodes.reserve(tris.size() * 2);
106 build_recursive(nodes, tris, refs, 0, static_cast<int>(refs.size()), 8);
107
108 // Reorder triangles to match leaf ranges for better locality.
109 std::vector<CPU_Triangle> remapped;
110 remapped.reserve(tris.size());
111
112 // Simple DFS stack to iterate nodes without recursion.
113 std::vector<int> stack;
114 stack.push_back(0);
115
116 while (!stack.empty()) {
117 const int n = stack.back();
118 stack.pop_back();
119 const auto &node = nodes[n];
120
121 if (node.isLeaf()) {
122 // Pack this leaf's triangles contiguously.
123 for (int i = 0; i < node.count; ++i) {
124 remapped.push_back(tris[refs[node.first + i].triIndex]);
125 }
126 const int base = static_cast<int>(remapped.size()) - node.count;
127 // Store the base index into the remapped array.
128 nodes[n].first = base;
129 } else {
130 stack.push_back(node.left);
131 stack.push_back(node.right);
132 }
133 }
134
135 tris = std::move(remapped);
136 return nodes;
137}

223// -------- Extract triangles from Model -----------
224// Flattens a LearnOpenGL-style Model into CPU_Triangle list, applying M.
225void gather_model_triangles(const Model &model,
226 const glm::mat4 &M,
227 std::vector<CPU_Triangle> &outTris) {
228 // Assumes:
229 // mesh.vertices[i].Position (glm::vec3)
230 // mesh.indices (uint32_t triplets)
231 for (const auto &mesh: model.meshes) {
232 const auto &V = mesh.vertices;
233 const auto &I = mesh.indices;
234 for (size_t k = 0; k + 2 < I.size(); k += 3) {
235 auto p0 = glm::vec3(M * glm::vec4(V[I[k]].Position, 1.0f));
236 auto p1 = glm::vec3(M * glm::vec4(V[I[k + 1]].Position, 1.0f));
237 auto p2 = glm::vec3(M * glm::vec4(V[I[k + 2]].Position, 1.0f));
238
239 CPU_Triangle T{};
240 T.v0 = p0;
241 T.e1 = p1 - p0;
242 T.e2 = p2 - p0;
243 outTris.push_back(T);
244 }
245 }
246}
