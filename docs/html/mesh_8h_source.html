1/*
2 Mesh Class
3
4 This class manages the GPU representation of a mesh using modern OpenGL.
5 It handles:
6
7 - VAO (Vertex Array Object): remembers how vertex data is laid out in memory.
8 - VBO (Vertex Buffer Object): stores vertex attributes like positions and normals.
9 - EBO (Element Buffer Object): stores indices that define faces via shared vertices.
10
11 Once initialized, this class allows the mesh to be rendered via a simple .Draw() call.
12
13 References:
14 - https://learnopengl.com/Getting-started/Hello-Triangle
15 - RAII Principles: https://en.cppreference.com/w/cpp/language/raii
16 - Based on: https://github.com/JoeyDeVries/LearnOpenGL/blob/master/includes/learnopengl/mesh.h
17
18 Design Notes:
19 - Mesh is a **move-only** class. Copying is disabled to avoid multiple owners of GPU resources.
20 - It follows RAII: resources (VBO, VAO, EBO) are acquired and released automatically.
21 - No texturing is implemented in this version.
22*/

170 void setupMesh() {
171 glGenVertexArrays(1, &VAO);
172 glGenBuffers(1, &VBO);
173 glGenBuffers(1, &EBO);
174
175 glBindVertexArray(VAO);
176
177 glBindBuffer(GL_ARRAY_BUFFER, VBO);
178 glBufferData(GL_ARRAY_BUFFER,
179 static_cast<GLsizeiptr>(sizeof(Vertex) * vertices.size()),
180 vertices.data(), GL_STATIC_DRAW);
181
182 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
183 glBufferData(GL_ELEMENT_ARRAY_BUFFER,
184 static_cast<GLsizeiptr>(sizeof(GLuint) * indices.size()),
185 indices.data(), GL_STATIC_DRAW);
186
187 // Position attribute (layout = 0)
188 glEnableVertexAttribArray(0);
189 glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
190 static_cast<void *>(nullptr));
191
192 // Normal attribute (layout = 1)
193 glEnableVertexAttribArray(1);
194 glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
195 reinterpret_cast<void *>(offsetof(Vertex, Normal)));
196
197 // Texture coordinates (layout = 2)
198 glEnableVertexAttribArray(2);
199 glVertexAttribPointer(2, 2, GL_FLOAT, GL_FALSE, sizeof(Vertex),
200 reinterpret_cast<void *>(offsetof(Vertex, TexCoords)));
201
202 // Tangent (layout = 3)
203 glEnableVertexAttribArray(3);
204 glVertexAttribPointer(3, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
205 reinterpret_cast<void *>(offsetof(Vertex, Tangent)));
206
207 // Bitangent (layout = 4)
208 glEnableVertexAttribArray(4);
209 glVertexAttribPointer(4, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex),
210 reinterpret_cast<void *>(offsetof(Vertex, Bitangent)));
211
212 glBindBuffer(GL_ARRAY_BUFFER, 0);
213 glBindVertexArray(0); // EBO stays bound to VAO
