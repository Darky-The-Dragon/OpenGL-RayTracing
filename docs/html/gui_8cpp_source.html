14 // Simple in-UI debug console (based on ImGui demo)
15 // ============================================================================
16 struct DebugConsole {
17 ImGuiTextBuffer Buf;
18 ImGuiTextFilter Filter;
19 ImVector<int> LineOffsets; // Index to lines in Buf
20 bool AutoScroll = true;
21
22 DebugConsole() {
23 Clear();
24 }
25
26 // Reset console contents and line offsets.
27 void Clear() {
28 Buf.clear();
29 LineOffsets.clear();
30 LineOffsets.push_back(0);
31 }
32
33 // Append a formatted line to the console buffer.
34 void AddLog(const char *fmt, ...) IM_FMTARGS(2) {
35 const int old_size = Buf.size();
36
37 va_list args;
38 va_start(args, fmt);
39 Buf.appendfv(fmt, args);
40 va_end(args);
41
42 // Track line starts for fast filtering / scrolling.
43 for (int i = old_size; i < Buf.size(); ++i) {
44 if (Buf[i] == '\n')
45 LineOffsets.push_back(i + 1);
46 }
47 }
48
49 // Draw the console window (with filter + scrolling region).
50 void Draw(const char *title, bool *pOpen) {
51 constexpr ImGuiWindowFlags flags =
52 ImGuiWindowFlags_NoCollapse |
53 ImGuiWindowFlags_NoSavedSettings |
54 ImGuiWindowFlags_NoFocusOnAppearing;
55
56 if (!ImGui::Begin(title, pOpen, flags)) {
57 ImGui::End();
58 return;
59 }
60
61 // Top bar: clear + filter
62 if (ImGui::Button("Clear"))
63 Clear();
64 ImGui::SameLine();
65 Filter.Draw("Filter");
66 ImGui::Separator();
67
68 // Scrolling region
69 ImGui::BeginChild(
70 "scrolling",
71 ImVec2(0, 0),
72 false,
73 ImGuiWindowFlags_HorizontalScrollbar
74 );
75 ImGui::PushStyleVar(ImGuiStyleVar_ItemSpacing, ImVec2(0, 0));
76
77 const char *buf = Buf.begin();
78 const char *buf_end = Buf.end();
79
80 if (Filter.IsActive()) {
81 // Display only lines that pass filter
82 for (int line_no = 0; line_no < LineOffsets.Size; ++line_no) {
83 const char *line_start = buf + LineOffsets[line_no];
84 const char *line_end = (line_no + 1 < LineOffsets.Size)
85 ? (buf + LineOffsets[line_no + 1] - 1)
86 : buf_end;
87 if (Filter.PassFilter(line_start, line_end))
88 ImGui::TextUnformatted(line_start, line_end);
89 }
90 } else {
91 // Display everything
92 ImGui::TextUnformatted(buf, buf_end);
93 }
94
95 ImGui::PopStyleVar();
96
97 // Auto-scroll to bottom if already at bottom.
98 if (AutoScroll && ImGui::GetScrollY() >= ImGui::GetScrollMaxY())
99 ImGui::SetScrollHereY(1.0f);
100
101 ImGui::EndChild();
102 ImGui::End();
103 }
104 };

132 // ============================================================================
133 // Log: mirror to terminal + GUI console
134 // ============================================================================
135 void Log(const char *fmt, ...) {
136 char buf[1024];
137
138 va_list args;
139 va_start(args, fmt);
140 std::vsnprintf(buf, sizeof(buf), fmt, args);
141 va_end(args);
142
143 // Terminal output
144 std::cout << buf;
145 std::cout.flush();
146
147 // GUI console output
148 gConsole.AddLog("%s", buf);
149 }

694 bool &useBVH,
695 bool &showMotion,
696 BvhModelPickerState &bvhPicker,
697 EnvMapPickerState &envPicker) {
698 // --------------------------------------------------------------
699 // Disable ALL ImGui mouse input when scene input (captured mouse) is active.
700 // This prevents hovering, clicking, tooltips, highlights, etc.
701 // --------------------------------------------------------------
702 if (input.sceneInputEnabled) {
703 ImGuiIO &io = ImGui::GetIO();
704
705 // Move the mouse far away so nothing is hovered
706 io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
707
708 // Disable mouse buttons so UI cannot click anything
709 for (bool &i: io.MouseDown)
710 i = false;
711
712 // Ensure wheel scrolling doesn't affect UI
713 io.MouseWheel = 0.0f;
714 io.MouseWheelH = 0.0f;
715 }
716
717 DrawMainControls(params, frame, input, rayMode, useBVH, showMotion);
718 DrawKeybindLegend();
719
720 // --------------------------------------------------------------------
721 // BVH model picker (top-right) only visible when BVH is enabled
722 // --------------------------------------------------------------------
723 if (useBVH) {
724 // Scan models directory for .obj files once (or when forced).
725 if (!gModelScanDone) {
726 namespace fs = std::filesystem;
727 gModelDir = util::resolve_dir("models");
728 gModelFiles.clear();
729 try {
730 for (const auto &entry: fs::directory_iterator(gModelDir)) {
731 if (!entry.is_regular_file()) continue;
732 const auto &p = entry.path();
733 if (p.extension() == ".obj") {
734 gModelFiles.push_back(p.string());
735 }
736 }
737 } catch (const std::exception &e) {
738 Log("[BVH GUI] Failed to scan '%s': %s\n", gModelDir.c_str(), e.what());
739 }
740
741 if (bvhPicker.selectedIndex >= static_cast<int>(gModelFiles.size())) {
742 bvhPicker.selectedIndex = 0;
743 }
744
745 if (!gModelFiles.empty()) {
746 std::snprintf(bvhPicker.currentPath,
747 sizeof(bvhPicker.currentPath),
748 "%s",
749 gModelFiles[bvhPicker.selectedIndex].c_str());
750 }
751
752 gModelScanDone = true;
753 }
754
755 const ImGuiViewport *vp = ImGui::GetMainViewport();
756 const ImVec2 pos(vp->WorkPos.x + vp->WorkSize.x - 10.0f,
757 vp->WorkPos.y + 10.0f);
758
759 ImGui::SetNextWindowPos(pos, ImGuiCond_Always, ImVec2(1.0f, 0.0f));
760
761 constexpr ImGuiWindowFlags pickerFlags =
762 ImGuiWindowFlags_NoResize |
763 ImGuiWindowFlags_AlwaysAutoResize |
764 ImGuiWindowFlags_NoSavedSettings |
765 ImGuiWindowFlags_NoCollapse;
766
767 if (ImGui::Begin("BVH Model Picker", nullptr, pickerFlags)) {
768 ImGui::Text("Models in %s/", gModelDir.c_str());
769 ImGui::Separator();
770
771 if (gModelFiles.empty()) {
772 ImGui::TextUnformatted("No .obj files found.");
773 } else {
774 for (int i = 0; i < static_cast<int>(gModelFiles.size()); ++i) {
775 const bool isSelected = (i == bvhPicker.selectedIndex);
776 const char *label = gModelFiles[i].c_str();
777 if (ImGui::Selectable(label, isSelected)) {
778 if (!isSelected) {
779 bvhPicker.selectedIndex = i;
780 std::snprintf(bvhPicker.currentPath,
781 sizeof(bvhPicker.currentPath),
782 "%s",
783 gModelFiles[i].c_str());
784 bvhPicker.reloadRequested = true;
785 Log("[BVH GUI] Selected model: %s\n",
786 bvhPicker.currentPath);
787 }
788 }
789 }
790 }
791
792 if (ImGui::Button("Rescan folder")) {
793 gModelScanDone = false; // trigger rescan next frame
794 Log("[BVH GUI] Rescanning '%s'...\n", gModelDir.c_str());
795 }
796
797 ImGui::Separator();
798 ImGui::TextWrapped("Current: %s", bvhPicker.currentPath);
799 }
800 ImGui::End();
801 }
802
803 // --------------------------------------------------------------------
804 // Env Map picker (top-right, under BVH picker)
805 // --------------------------------------------------------------------
806 {
807 // Scan cubemap directory for supported image formats once (or when forced).
808 if (!gEnvScanDone) {
809 namespace fs = std::filesystem;
810 gEnvDir = util::resolve_dir("cubemaps");
811 gEnvFiles.clear();
812 try {
813 for (const auto &entry: fs::directory_iterator(gEnvDir)) {
814 if (!entry.is_regular_file()) continue;
815 const auto &p = entry.path();
816 auto ext = p.extension().string();
817 if (ext == ".png" || ext == ".jpg" ||
818 ext == ".jpeg" || ext == ".hdr" ||
819 ext == ".exr") {
820 gEnvFiles.push_back(p.string());
821 }
822 }
823 } catch (const std::exception &e) {
824 Log("[ENV GUI] Failed to scan '%s': %s\n",
825 gEnvDir.c_str(), e.what());
826 }
827
828 if (envPicker.selectedIndex >= static_cast<int>(gEnvFiles.size())) {
829 envPicker.selectedIndex = 0;
830 }
831
832 if (!gEnvFiles.empty()) {
833 std::snprintf(envPicker.currentPath,
834 sizeof(envPicker.currentPath),
835 "%s",
836 gEnvFiles[envPicker.selectedIndex].c_str());
837 }
838
839 gEnvScanDone = true;
840 }
841
842 const ImGuiViewport *vp = ImGui::GetMainViewport();
843 // Slightly below the BVH picker (offset y by ~120 px)
844 const ImVec2 pos(vp->WorkPos.x + vp->WorkSize.x - 10.0f,
845 vp->WorkPos.y + 130.0f);
846
847 ImGui::SetNextWindowPos(pos, ImGuiCond_Always, ImVec2(1.0f, 0.0f));
848
849 constexpr ImGuiWindowFlags pickerFlags =
850 ImGuiWindowFlags_NoResize |
851 ImGuiWindowFlags_AlwaysAutoResize |
852 ImGuiWindowFlags_NoSavedSettings |
853 ImGuiWindowFlags_NoCollapse;
854
855 if (ImGui::Begin("Env Map Picker", nullptr, pickerFlags)) {
856 ImGui::Text("Cubemaps in %s/", gEnvDir.c_str());
857 ImGui::Separator();
858
859 if (gEnvFiles.empty()) {
860 ImGui::TextUnformatted("No cubemap images found.");
861 } else {
862 for (int i = 0; i < static_cast<int>(gEnvFiles.size()); ++i) {
863 const bool isSelected = (i == envPicker.selectedIndex);
864 const char *label = gEnvFiles[i].c_str();
865 if (ImGui::Selectable(label, isSelected)) {
866 if (!isSelected) {
867 envPicker.selectedIndex = i;
868 std::snprintf(envPicker.currentPath,
869 sizeof(envPicker.currentPath),
870 "%s",
871 gEnvFiles[i].c_str());
872 envPicker.reloadRequested = true;
873 Log("[ENV GUI] Selected env map: %s\n",
874 envPicker.currentPath);
875 }
876 }
877 }
878 }
879
880 if (ImGui::Button("Rescan folder")) {
881 gEnvScanDone = false; // trigger rescan next frame
882 Log("[ENV GUI] Rescanning '%s'...\n", gEnvDir.c_str());
883 }
884
885 ImGui::Separator();
886 ImGui::TextWrapped("Current: %s", envPicker.currentPath);
887 }
888 ImGui::End();
889 }
890
891 // Big, wide console pinned bottom-left
892 if (gShowDebugConsole) {
893 const ImGuiViewport *vp = ImGui::GetMainViewport();
894 const ImVec2 pos(vp->WorkPos.x + 10.0f,
895 vp->WorkPos.y + vp->WorkSize.y - 10.0f);
896
897 ImGui::SetNextWindowPos(pos, ImGuiCond_Always, ImVec2(0.0f, 1.0f));
898 ImGui::SetNextWindowSize(ImVec2(700.0f, 260.0f), ImGuiCond_Always);
899 gConsole.Draw("Debug Console", &gShowDebugConsole);
900 }
901 }
