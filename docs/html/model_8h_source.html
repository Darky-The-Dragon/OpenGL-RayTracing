1/*
2 Model class
3
4 Loads .obj models using the Assimp library and converts the imported data into a format
5 compatible with OpenGL rendering via the Mesh class.
6
7 Key Concepts:
8 - Uses Assimp to parse 3D model files
9 - Converts Assimp mesh data to OpenGL-compatible structures
10 - Stores each sub-mesh as a Mesh instance
11 - Delegates rendering to Mesh::Draw()
12
13 Notes:
14 1) Follows RAII: Model and Mesh manage their own GPU resources and cleanup automatically.
15 2) Copying is disabled — this is a move-only class to avoid duplicating GPU resources.
16 3) This version does not support textures.
17 4) Based on LearnOpenGL’s model loading example by Joey de Vries.

19 Authors: Davide Gadia, Michael Marchesan

105 void loadModel(const std::string &path) {
106 Assimp::Importer importer;
107
108 const aiScene *scene = importer.ReadFile(
109 path,
110 aiProcess_Triangulate |
111 aiProcess_JoinIdenticalVertices |
112 aiProcess_FlipUVs |
113 aiProcess_GenSmoothNormals |
114 aiProcess_CalcTangentSpace
115 );
116
117 if (!scene || (scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE) || !scene->mRootNode) {
118 std::cerr << "ERROR::ASSIMP:: " << importer.GetErrorString() << '\n';
119 return;
120 }
121
122 processNode(scene->mRootNode, scene);
123 }

134 void processNode(const aiNode *node, const aiScene *scene) {
135 for (unsigned int i = 0; i < node->mNumMeshes; ++i) {
136 aiMesh *mesh = scene->mMeshes[node->mMeshes[i]];
137 meshes.emplace_back(processMesh(mesh));
138 }
139
140 for (unsigned int i = 0; i < node->mNumChildren; ++i)
141 processNode(node->mChildren[i], scene);
142 }

156 static Mesh processMesh(aiMesh *mesh) {
157 std::vector<Vertex> vertices;
158 std::vector<GLuint> indices;
159
160 bool warnedNoUV = false;
161
162 vertices.reserve(mesh->mNumVertices);
163 for (unsigned int i = 0; i < mesh->mNumVertices; ++i) {
164 Vertex vertex{};
165
166 // Position
167 vertex.Position = {
168 mesh->mVertices[i].x,
169 mesh->mVertices[i].y,
170 mesh->mVertices[i].z
171 };
172
173 // Normal
174 if (mesh->HasNormals()) {
175 vertex.Normal = {
176 mesh->mNormals[i].x,
177 mesh->mNormals[i].y,
178 mesh->mNormals[i].z
179 };
180 } else {
181 vertex.Normal = glm::vec3(0.0f, 1.0f, 0.0f);
182 }
183
184 // Texture coordinates, tangent, bitangent
185 if (mesh->mTextureCoords[0]) {
186 vertex.TexCoords = {
187 mesh->mTextureCoords[0][i].x,
188 mesh->mTextureCoords[0][i].y
189 };
190 if (mesh->mTangents && mesh->mBitangents) {
191 vertex.Tangent = {
192 mesh->mTangents[i].x,
193 mesh->mTangents[i].y,
194 mesh->mTangents[i].z
195 };
196 vertex.Bitangent = {
197 mesh->mBitangents[i].x,
198 mesh->mBitangents[i].y,
199 mesh->mBitangents[i].z
200 };
201 } else {
202 vertex.Tangent = glm::vec3(0.0f);
203 vertex.Bitangent = glm::vec3(0.0f);
204 }
205 } else {
206 vertex.TexCoords = glm::vec2(0.0f);
207 vertex.Tangent = glm::vec3(0.0f);
208 vertex.Bitangent = glm::vec3(0.0f);
209 if (!warnedNoUV) {
210 std::cout << "WARNING: Model lacks UVs — Tangent/Bitangent set to 0.\n";
211 warnedNoUV = true;
212 }
213 }
214
215 vertices.emplace_back(vertex);
216 }
217
218 // Read all face indices
219 indices.reserve(mesh->mNumFaces * 3);
220 for (unsigned int i = 0; i < mesh->mNumFaces; ++i) {
221 aiFace &face = mesh->mFaces[i];
222 for (unsigned int j = 0; j < face.mNumIndices; ++j)
223 indices.emplace_back(face.mIndices[j]);
224 }
225
226 // Ownership and GPU upload occur inside Mesh
227 return Mesh{std::move(vertices), std::move(indices)};
228 }
