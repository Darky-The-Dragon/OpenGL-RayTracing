33// Load a cubemap from a single 4x3 cross-layout image.
34// The function extracts the 6 faces and uploads them to a new cubemap texture.
35GLuint loadCubeMapFromCross(const std::string &path) {
36 int width, height, channels;
37
38 // Cubemap cross must not be vertically flipped
39 stbi_set_flip_vertically_on_load(0);
40 unsigned char *data = stbi_load(path.c_str(), &width, &height, &channels, 0);
41 if (!data) {
42 fprintf(stderr, "Failed to load cubemap cross image: %s\n", path.c_str());
43 return 0;
44 }
45
46 // Validate cross layout: 4 tiles wide, 3 tiles tall
47 if ((height % 3) != 0 || (width % 4) != 0 || (width / 4) != (height / 3)) {
48 fprintf(stderr,
49 "Invalid cubemap cross dimensions: %dx%d (expected 4x3 tiles)\n",
50 width, height);
51 stbi_image_free(data);
52 return 0;
53 }

55 const int faceSize = height / 3; // Size of each cube face
56 const int stride = width * channels; // Bytes per image row
57 const GLenum format = (channels == 4) ? GL_RGBA : GL_RGB;
58 const GLenum internalFormat = (channels == 4) ? GL_RGBA8 : GL_RGB8;
59
60 GLuint texID = 0;
61 glGenTextures(1, &texID);
62 glBindTexture(GL_TEXTURE_CUBE_MAP, texID);
63
64 std::vector<unsigned char> faceBuf(faceSize * faceSize * channels);
65
66 // Helper to upload a single face from position (ox, oy) in the cross
67 auto uploadFace = [&](GLenum faceTarget, int ox, int oy) {
68 for (int y = 0; y < faceSize; ++y) {
69 const unsigned char *srcRow =
70 data + (oy + y) * stride + ox * channels;
71 unsigned char *dstRow =
72 faceBuf.data() + y * faceSize * channels;
73
74 memcpy(dstRow, srcRow, faceSize * channels);
75 }
76
77 glTexImage2D(faceTarget, 0,
78 internalFormat, faceSize, faceSize, 0,
79 format, GL_UNSIGNED_BYTE, faceBuf.data());

83 // [ ][ +Y][ ][ ]
84 // [-X][ +Z][ +X][ -Z]
85 // [ ][ -Y][ ][ ]
86 uploadFace(GL_TEXTURE_CUBE_MAP_POSITIVE_X, 2 * faceSize, 1 * faceSize);
87 uploadFace(GL_TEXTURE_CUBE_MAP_NEGATIVE_X, 0 * faceSize, 1 * faceSize);
88 uploadFace(GL_TEXTURE_CUBE_MAP_POSITIVE_Y, 1 * faceSize, 0 * faceSize);
89 uploadFace(GL_TEXTURE_CUBE_MAP_NEGATIVE_Y, 1 * faceSize, 2 * faceSize);
