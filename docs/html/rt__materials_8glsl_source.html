5/*
6 rt_materials.glsl – Material IDs and Parameter Fetch
7
8 This module defines:
9 - Symbolic material IDs for the analytic scene (floor + spheres).
10 - A compact MaterialProps struct used during shading.
11 - A getMaterial(id) function that maps a material ID to concrete
12 parameters, many of which are driven by GUI-controlled uniforms.
13 - A convenience helper materialAlbedo(id) for quick albedo queries.
14
15 The IDs here must stay in sync with the analytic scene geometry setup
16 on the CPU side (rt_scene_analytic.glsl + C++ scene layout).
17*/
18
19// Material IDs (must match analytic scene)
20const int MAT_FLOOR = 0;
21const int MAT_ALBEDO_SPHERE = 1;
22const int MAT_GLASS_SPHERE = 2;
23const int MAT_MIRROR_SPHERE = 3;
24const int MAT_POINTLIGHT_SPHERE = 4;
25
26/**
27 * @brief Material parameter block used by shading code.
28 *
29 * Fields:
30 * - albedo : base color (diffuse / tint)
31 * - specStrength : specular term weight (Phong/other)
32 * - gloss : specular exponent or glossiness factor
33 * - type : 0 = lambert/Phong, 1 = mirror, 2 = glass
34 * - ior : index of refraction for glass materials
35 */
36struct MaterialProps {
37 vec3 albedo;
38 float specStrength;
39 float gloss;
40 int type; // 0 = lambert, 1 = mirror, 2 = glass
41 float ior; // index of refraction for glass
42};
43
44/**
45 * @brief Returns material properties for a given material ID.
46 *
47 * The mapping is:
48 * - MAT_FLOOR : static neutral grey floor
49 * - MAT_ALBEDO_SPHERE : GUI-driven base sphere (albedo/spec/gloss)
50 * - MAT_GLASS_SPHERE : glass sphere (tint + IOR), can be disabled → diffuse
51 * - MAT_MIRROR_SPHERE : mirror sphere, can be disabled → diffuse fallback
52 * - MAT_POINTLIGHT_SPHERE: emissive sphere – typically handled specially
53 *
54 * Many parameters are controlled via uniforms so that the GUI can tweak
55 * them in real time.
56 */
57MaterialProps getMaterial(int id) {
58 MaterialProps m;
59
60 // Floor – fixed neutral grey
61 if (id == MAT_FLOOR) {
62 m.albedo = vec3(0.7);
63 m.specStrength = 0.1;
64 m.gloss = 16.0;
65 m.type = 0;
66 m.ior = 1.0;
67 return m;
68 }
69
70 // Left sphere – GUI albedo
71 if (id == MAT_ALBEDO_SPHERE) {
72 m.albedo = uMatAlbedo_AlbedoColor;
73 m.specStrength = uMatAlbedo_SpecStrength;
74 m.gloss = uMatAlbedo_Gloss;
75 m.type = 0;
76 m.ior = 1.0;
77 return m;
78 }
79
80 // Glass sphere
81 if (id == MAT_GLASS_SPHERE) {
82 if (uMatGlass_Enabled == 0) {
83 // behave like diffuse if disabled
84 m.albedo = uMatAlbedo_AlbedoColor;
85 m.specStrength = uMatAlbedo_SpecStrength;
86 m.gloss = uMatAlbedo_Gloss;
87 m.type = 0;
88 m.ior = 1.0;
89 } else {
90 m.albedo = uMatGlass_Albedo;
91 m.specStrength = uMatGlass_Distortion; // used by shadeGlass as distortion strength
92 m.gloss = 1.0;
93 m.type = 2;
94 m.ior = uMatGlass_IOR;
95 }
96 return m;
97 }
98
99 // Mirror sphere
100 if (id == MAT_MIRROR_SPHERE) {
101 if (uMatMirror_Enabled == 0) {
102 // fallback to diffuse if disabled
103 m.albedo = uMatAlbedo_AlbedoColor;
104 m.specStrength = uMatAlbedo_SpecStrength;
105 m.gloss = uMatAlbedo_Gloss;
106 m.type = 0;
107 m.ior = 1.0;
108 } else {
109 m.albedo = uMatMirror_Albedo;
110 m.specStrength = 0.0;
111 m.gloss = uMatMirror_Gloss;
112 m.type = 1; // mirror
113 m.ior = 1.0;
114 }
115 return m;
116 }
117
118 // Fallback (e.g. pointlight sphere or unknown ID)
119 m.albedo = vec3(0.8);
120 m.specStrength = 0.2;
121 m.gloss = 16.0;
122 m.type = 0;
123 m.ior = 1.0;
124 return m;
125}

127/**
128 * @brief Convenience helper returning only the albedo of a material.
129 */
130vec3 materialAlbedo(int id) {
131 return getMaterial(id).albedo;
132}
