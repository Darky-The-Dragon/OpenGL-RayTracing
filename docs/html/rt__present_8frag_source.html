3/*
4 rt_present.frag – TAA + SVGF Present / Post-Processing Shader
5
6 This shader:
7 - Reads the history buffer (uTex), which stores:
8 * rgb = accumulated linear color
9 * a = M2 (second moment of luma) used to estimate variance.
10 - Uses the motion buffer (uMotionTex) to:
11 * Visualize motion (debug mode), or
12 * Drive a variance- and motion-aware SVGF-lite spatial filter.
13 - Optionally applies a GBuffer-aware spatial filter (svgfFilter):
14 * Uses world-space position (uGPos) and normal (uGNrm) to stop blurring
15 across geometry edges.
16 - Applies ACES tonemapping and gamma correction to output sRGB.
17
18 Controls:
19 - uShowMotion: switches between normal path and motion visualization.
20 - uEnableSVGF: toggles the SVGF-lite filter.
21 - uSvgfStrength: blends between raw TAA output and filtered result.
22 - uKVar/uKColor, uKVarMotion/uKColorMotion: tuning parameters for the filter.
23
24 The shader is intended as a lightweight, real-time friendly post-process
25 that builds on top of the ray-traced accumulation + TAA pass.
26*/
27
28in vec2 vUV; // kept for compatibility, not used for sampling
29out vec4 fragColor;

31uniform sampler2D uTex; // history buffer: rgb = color, a = M2 (second moment of luma)
32uniform sampler2D uMotionTex; // RG16F, NDC motion (currNDC - prevNDC)
33
34// GBuffer samplers
35uniform sampler2D uGPos; // world-space position
36uniform sampler2D uGNrm; // world-space normal
37
38uniform float uExposure;
39uniform int uShowMotion; // 0 = normal, 1 = visualize motion
40uniform float uMotionScale; // e.g. 4.0
41uniform vec2 uResolution; // framebuffer size in pixels

46uniform float uKColor;

52uniform int uEnableSVGF;

54// Luma coefficients (approx. Rec.709)
55const vec3 YCOEFF = vec3(0.299, 0.587, 0.114);

57// -----------------------------------------------------------------------------
58// Tonemapping and color utilities
59// -----------------------------------------------------------------------------
60
61/**
62 * @brief ACES approximation tonemapper (Narkowicz 2015).
63 *
64 * Applies exposure, then maps HDR linear color into [0,1] with a curve
65 * that maintains highlight rolloff and overall contrast.
66 */
67vec3 acesTonemap(vec3 x) {
68 x *= uExposure;
69 const float a = 2.51, b = 0.03, c = 2.43, d = 0.59, e = 0.14;
70 return clamp((x * (a * x + b)) / (x * (c * x + d) + e), 0.0, 1.0);
71}

73/**
74 * @brief Converts HSV to RGB, mainly used for motion visualization.
75 */
76vec3 hsv2rgb(vec3 c) {
77 vec3 p = abs(fract(c.xxx + vec3(0.0, 2.0 / 3.0, 1.0 / 3.0)) * 6.0 - 3.0);
78 return c.z * mix(vec3(1.0), clamp(p - 1.0, 0.0, 1.0), c.y);
79}

81// ------------------------------------------------------------
82// Motion visualization (debug)
83// ------------------------------------------------------------
84
85/**
86 * @brief Visualizes 2D motion vectors as colored arrows.
87 *
88 * The hue encodes direction, and the value encodes magnitude (after scaling).
89 * Very small motions fall into a deadband and are shown as black.
90 *
91 * @param motion NDC motion vector (currNDC - prevNDC).
92 * @param scale Scalar used to amplify the vector before visualization.
93 */
94vec3 visualizeMotion(vec2 motion, float scale) {
95 vec2 m = motion * scale;
96
97 float mag = length(m);
98 if (mag < 1e-4) {
99 return vec3(0.0); // deadband → black
100 }
101
102 float hue = atan(m.y, m.x) / (2.0 * 3.1415926535) + 0.5;
103 float val = clamp(mag, 0.0, 1.0);
104
105 return hsv2rgb(vec3(hue, 1.0, val));
106}

108// ------------------------------------------------------------
109// SVGF-lite spatial filter using VARIANCE derived from M2 in alpha
110// Motion-aware + GBUFFER-aware (position + normal)
111// ------------------------------------------------------------
112
113/**
114 * @brief SVGF-lite spatial filter around the current pixel.
115 *
116 * Uses:
117 * - M2-based variance from the history buffer (uTex) to estimate noise.
118 * - Motion magnitude to modulate filter strength (more blur on moving content).
119 * - GBuffer-based edge-stopping (position + normal) to preserve geometry edges.
120 *
121 * It operates on a 3x3 neighborhood and computes weights based on:
122 * - variance (wVar),
123 * - color difference (wCol),
124 * - position difference (wPos),
125 * - normal difference (wNrm),
126 * - a small spatial bias (wSpatial) that favors the center sample.
127 */
128vec3 svgfFilter(vec2 uv) {
129 // Center sample: color + M2 of luma stored in A
130 vec4 centerRaw = texture(uTex, uv);
131 vec3 cCenter = centerRaw.rgb;
132 float M2center = centerRaw.a;
133
134 // Derive variance at center: Var = M2 - (E[l])^2
135 float lCenter = dot(cCenter, YCOEFF);
136 float varCenter = max(M2center - lCenter * lCenter, 0.0);
137
138 // Motion at center pixel
139 vec2 motion = texture(uMotionTex, uv).xy;
140 float motMag = length(motion);
141
142 // GBuffer at center
143 vec3 pCenter = texture(uGPos, uv).xyz;
144 vec3 nCenter = texture(uGNrm, uv).xyz;
145
146 // Clamp variance to a reasonable max – prevents insane weights
147 varCenter = min(varCenter, uVarMax);
148
149 // If variance is essentially tiny *and* motion is tiny → skip blur.
150 if (varCenter < uSvgfVarStaticEps && motMag < uSvgfMotionStaticEps) {
151 return cCenter;
152 }
153
154 vec2 texel = 1.0 / uResolution;
155
156 vec3 accumCol = vec3(0.0);
157 float accumW = 0.0;
158
159 // ----------------------------------------
160 // Motion-aware filter strength, driven by CPU params
161 // uKVar / uKColor control the *overall* sharpness vs blur.
162 // ----------------------------------------
163
164 // t = 0 → static, t = 1 → fully moving
165 float t = clamp(smoothstep(0.005, 0.05, motMag), 0.0, 1.0);
166
167 float kVar = mix(uKVar, uKVarMotion, t);
168 float kColor = mix(uKColor, uKColorMotion, t);
169
170 // Geometry weights – fixed scalars for now
171 const float K_NRM = 32.0; // higher = stronger normal edge stopping
172 const float K_POS = 1.0; // units are ~1 / (worldUnit^2)
173
174 for (int j = -1; j <= 1; ++j) {
175 for (int i = -1; i <= 1; ++i) {
176 vec2 offs = vec2(i, j);
177 vec2 uvN = uv + offs * texel;
178
179 // Avoid sampling outside screen
180 if (uvN.x < 0.0 || uvN.x > 1.0 ||
181 uvN.y < 0.0 || uvN.y > 1.0) {
182 continue;
183 }
184
185 vec4 s = texture(uTex, uvN);
186 vec3 c = s.rgb;
187 float M2 = s.a;
188
189 float l = dot(c, YCOEFF);
190 float v = max(M2 - l * l, 0.0);
191 v = min(v, uVarMax);
192
193 float wVar = exp(-v * kVar);
194
195 vec3 dc = c - cCenter;
196 float dc2 = dot(dc, dc);
197 float wCol = exp(-dc2 * kColor);
198
199 // GBuffer-based edge stopping
200 vec3 p = texture(uGPos, uvN).xyz;
201 vec3 n = texture(uGNrm, uvN).xyz;
202
203 // Position distance
204 vec3 dp = p - pCenter;
205 float dist2 = dot(dp, dp);
206 float wPos = exp(-dist2 * K_POS);
207
208 // Normal difference
209 float ndot = clamp(dot(normalize(nCenter), normalize(n)), -1.0, 1.0);
210 float nDiff = max(0.0, 1.0 - ndot); // 0 when aligned, 1 when opposite
211 float wNrm = exp(-nDiff * K_NRM);
212
213 // Slightly favor center when static; when moving, neighbors matter more
214 float wSpatial = (i == 0 && j == 0)
215 ? 1.0
216 : mix(0.8, 1.1, t);
217
218 float w = wVar * wCol * wPos * wNrm * wSpatial;
219 accumCol += c * w;
220 accumW += w;
221 }
222 }
223
224 if (accumW <= 0.0)
225 return cCenter;
226
227 return accumCol / accumW;
228}

230// -----------------------------------------------------------------------------
231// Main
232// -----------------------------------------------------------------------------
233
234void main() {
235 ivec2 sz = textureSize(uTex, 0);
236 vec2 uv = (gl_FragCoord.xy + vec2(0.5)) / vec2(sz);
237
238 // Motion debug mode
239 if (uShowMotion == 1) {
240 vec2 m = texture(uMotionTex, uv).xy;
241 vec3 rgb = visualizeMotion(m, uMotionScale);
242 fragColor = vec4(rgb, 1.0);
243 return;
244 }
245
246 // Raw TAA result (no spatial filter)
247 vec3 raw = texture(uTex, uv).rgb;
248
249 vec3 linearColor;
250 if (uEnableSVGF == 0) {
251 // SVGF disabled → just use raw TAA result
252 linearColor = raw;
253 } else {
254 // SVGF-lite: variance-guided spatial filter with GBuffer edge stopping
255 vec3 filtered = svgfFilter(uv);
256
257 // Blend between raw and filtered based on uSvgfStrength
258 // - 0.0 → pure TAA (sharp, noisy)
259 // - 1.0 → full SVGF (smooth)
260 float s = clamp(uSvgfStrength, 0.0, 1.0);
261 linearColor = mix(raw, filtered, s);
262 }
263
264 // Tonemap + gamma
265 vec3 mapped = acesTonemap(linearColor);
266 vec3 outSRGB = pow(mapped, vec3(1.0 / 2.2));
267
268 fragColor = vec4(outSRGB, 1.0);
269}
