20 // Preprocess GLSL #include "file.glsl" recursively.
21 // Includes are resolved relative to baseDir.
22 std::string preprocessShaderSource(const std::string &source,
23 const std::string &baseDir,
24 int depth = 0) {
25 if (depth > 16) {
26 std::cerr << "[WARNING] Shader include depth > 16, possible include cycle. "
27 << "Aborting further includes.\n";
28 return source;
29 }
30
31 std::istringstream input(source);
32 std::ostringstream output;
33 std::string line;
34
35 while (std::getline(input, line)) {
36 // Trim leading whitespace to detect "#include" even if indented
37 size_t first_non_ws = line.find_first_not_of(" \t");
38 std::string trimmed = (first_non_ws == std::string::npos)
39 ? ""
40 : line.substr(first_non_ws);
41
42 // Handle #include "..."
43 if (trimmed.rfind("#include", 0) == 0) {
44 size_t q1 = trimmed.find('"');
45 size_t q2 = (q1 != std::string::npos)
46 ? trimmed.find('"', q1 + 1)
47 : std::string::npos;
48
49 if (q1 != std::string::npos && q2 != std::string::npos && q2 > q1 + 1) {
50 std::string incPath = trimmed.substr(q1 + 1, q2 - q1 - 1);
51 std::string fullPath = baseDir + incPath;
52 std::ifstream incFile(fullPath);
53 if (!incFile) {
54 std::cerr << "ERROR: Could not open included shader file: \""
55 << incPath << "\" (full path: " << fullPath << ")\n";
56 // Keep original line so the shader is still somewhat debuggable.
57 output << line << '\n';
58 continue;
59 }
60
61 std::stringstream incStream;
62 incStream << incFile.rdbuf();
63 std::string incSrc = incStream.str();
64
65 output << "// --- begin include: " << incPath << " ---\n";
66 output << preprocessShaderSource(
67 incSrc,
68 getDirectory(fullPath),
69 depth + 1
70 );
71 output << "// --- end include: " << incPath << " ---\n";
72 } else {
73 // Malformed include, just emit as-is
74 output << line << '\n';
75 }
76 } else {
77 output << line << '\n';
78 }
79 }
80 return output.str();
81 }

106Shader::Shader(const char *vertexPath, const char *fragmentPath) {
107 std::ifstream vFile(vertexPath);
108 std::ifstream fFile(fragmentPath);
109
110 if (!vFile || !fFile) {
111 std::cerr << "ERROR: Could not open shader files:\n"
112 << "Vertex: " << vertexPath << "\n"
113 << "Fragment: " << fragmentPath << "\n";
114 ID = 0;
115 valid = false;
116 return;
117 }
118
119 std::stringstream vStream, fStream;
120 vStream << vFile.rdbuf();
121 fStream << fFile.rdbuf();
122
123 std::string vRaw = vStream.str();
124 std::string fRaw = fStream.str();
125
126 // Expand #include "..." directives relative to each shader's directory.
127 std::string vCode = shader_detail::preprocessShaderSource(vRaw, shader_detail::getDirectory(vertexPath));
128 std::string fCode = shader_detail::preprocessShaderSource(fRaw, shader_detail::getDirectory(fragmentPath));
129
130 const char *vShaderCode = vCode.c_str();
131 const char *fShaderCode = fCode.c_str();
132
133 // Compile vertex shader
134 unsigned int vertex = glCreateShader(GL_VERTEX_SHADER);
135 glShaderSource(vertex, 1, &vShaderCode, nullptr);
136 glCompileShader(vertex);
137 checkCompileErrors(vertex, "VERTEX");
138
139 // Compile fragment shader
140 unsigned int fragment = glCreateShader(GL_FRAGMENT_SHADER);
141 glShaderSource(fragment, 1, &fShaderCode, nullptr);
142 glCompileShader(fragment);
143 checkCompileErrors(fragment, "FRAGMENT");
144
145 // Link shaders into a single program
146 ID = glCreateProgram();
147 glAttachShader(ID, vertex);
148 glAttachShader(ID, fragment);
149 glLinkProgram(ID);
150 checkCompileErrors(ID, "PROGRAM");
151
152 glDeleteShader(vertex);
153 glDeleteShader(fragment);
154
155 GLint linkStatus = 0;
156 glGetProgramiv(ID, GL_LINK_STATUS, &linkStatus);
157 valid = (linkStatus == GL_TRUE);
158}

202// Cached uniform location lookup. Returns -1 if program is invalid.
203int Shader::uniformLocation(const std::string &name) const {
204 if (ID == 0) return -1;
205
206 auto it = uniformCache.find(name);
207 if (it != uniformCache.end()) {
208 return it->second;
209 }
210
211 const int loc = glGetUniformLocation(ID, name.c_str());
212 uniformCache.emplace(name, loc);
213 return loc;
214}
