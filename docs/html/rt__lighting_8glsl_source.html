5/*
6 rt_lighting.glsl – Direct & Indirect Lighting for Analytic and BVH Scenes
7
8 This module defines:
9 - A simple disk area light (kLightCenter, kLightN, kLightRadius, kLightCol).
10 - Unified occlusion tests that work in both analytic and BVH modes.
11 - A shared Lambert + Phong BRDF helper.
12 - Sun, sky, and point lights (hybrid analytic lights shared across scenes).
13 - Direct lighting evaluators:
14 * directLight() – analytic scene (plane + spheres)
15 * directLightBVH() – BVH triangle scene
16 - One-bounce diffuse GI for analytic and BVH scenes with basic clamping.
17 - Glass shading with thin refraction and local reflections.
18 - Mirror shading using analytic scene traces.
19 - Ambient occlusion (AO) using cosine-weighted hemisphere sampling.
20
21 All routines assume the presence of:
22 - uUseBVH, uAO_* uniforms.
23 - traceAnalytic(), traceAnalyticIgnoreGlass(), traceAnalyticIgnorePointLight().
24 - traceBVH(), traceBVHShadow().
25 - MaterialProps, sky(), sampleHemisphereCosine(), etc.
26*/

28// ------------- Disk area light (existing) --------------
29const vec3 kLightCenter = vec3(0.0, 5.0, -3.0);
30const vec3 kLightN = normalize(vec3(0.0, -1.0, 0.2));
31const float kLightRadius = 1.2;

34// ---- Unified shadow test for both modes
35
36/**
37 * @brief Tests whether the segment between p and q is occluded.
38 *
39 * Builds a shadow ray from p toward q and checks:
40 * - BVH scene via traceBVHShadow, or
41 * - Analytic scene via traceAnalytic.
42 *
43 * @param p Start position (usually a surface point).
44 * @param q Target point (e.g., area-light point).
45 * @return True if any geometry blocks the path before reaching q.
46 */
47bool occludedToward(vec3 p, vec3 q) {
48 vec3 rd = normalize(q - p);
49 float maxT = length(q - p);
50 float eps = epsForDist(maxT);
51 if (uUseBVH == 1) {
52 return traceBVHShadow(p + rd * eps, rd, maxT - eps);
53 } else {
54 Hit h;
55 if (traceAnalytic(p + rd * eps, rd, h) && h.t < maxT - eps) return true;
56 return false;
57 }
58}

60// ---------------------------------------------------------------------------
61// Shared BRDF helper: Lambert + optional Phong spec
62// ---------------------------------------------------------------------------
63
64/**
65 * @brief Evaluates a Lambert + Phong BRDF for a single light sample.
66 *
67 * @param N Surface normal (world space, normalized).
68 * @param V View direction (from surface → camera).
69 * @param L Light direction (from surface → light).
70 * @param Li Incident radiance from the light.
71 * @param albedo Diffuse albedo color.
72 * @param specStrength Scalar multiplier for the specular lobe.
73 * @param gloss Phong exponent controlling highlight sharpness.
74 * @return Outgoing radiance contribution from this light sample.
75 */
76vec3 shadeLambertPhong(
77 vec3 N, vec3 V, vec3 L, vec3 Li,
78 vec3 albedo, float specStrength, float gloss)
79{
80 float ndl = max(dot(N, L), 0.0);
81 if (ndl <= 0.0) return vec3(0.0);
82
83 // Diffuse
84 vec3 diffuse = albedo * (ndl / uPI);
85
86 // Specular (Phong)
87 vec3 spec = vec3(0.0);
88 if (specStrength > 0.0) {
89 vec3 H = normalize(L + V);
90 float ndh = max(dot(N, H), 0.0);
91 float phong = pow(ndh, gloss);
92 spec = specStrength * phong * vec3(1.0);
93 }
94
95 return (diffuse + spec) * Li;
96}

98// ---------------------------------------------------------------------------
99// Sun light (directional, shadowed)
100// ---------------------------------------------------------------------------
101
102/**
103 * @brief Directional sunlight contribution with hard shadows.
104 *
105 * Uses the shared BRDF helper for non-glass/non-mirror materials.
106 *
107 * @param h Primary hit information.
108 * @param mat Material properties at the hit.
109 * @param Vdir Direction from hit → camera.
110 * @return Radiance contribution from the sun.
111 */
112vec3 sunDirect(Hit h, MaterialProps mat, vec3 Vdir)
113{
114 if (uSunEnabled == 0) return vec3(0.0);
115
116 vec3 N = normalize(h.n);
117 vec3 V = normalize(Vdir);
118 vec3 L = normalize(-uSunDir); // light dir from hit -> sun
119
120 float ndl = max(dot(N, L), 0.0);
121 if (ndl <= 0.0) return vec3(0.0);
122
123 // Shadow ray toward sun (approx "infinite" distance)
124 float maxT = 1000.0;
125 float eps = epsForDist(maxT);
126 vec3 origin = h.p + N * eps;
127
128 bool blocked;
129 if (uUseBVH == 1) {
130 blocked = traceBVHShadow(origin, L, maxT - eps);
131 } else {
132 Hit tmp;
133 blocked = traceAnalytic(origin, L, tmp);
134 }
135 if (blocked) return vec3(0.0);
136
137 vec3 Li = uSunColor * uSunIntensity;
138
139 // Only non-mirror/non-glass get Phong spec
140 float specStrength = (mat.type == 0) ? mat.specStrength : 0.0;
141 return shadeLambertPhong(N, V, L, Li, mat.albedo, specStrength, mat.gloss);
142}

144// ---------------------------------------------------------------------------
145// Sky dome (ambient-ish, no extra shadows; AO will handle occlusion feel)
146// ---------------------------------------------------------------------------
147
148/**
149 * @brief Simple hemispherical sky dome contribution.
150 *
151 * Approximates ambient light from a single sky direction uSkyUpDir with
152 * cosine-weighting. Shadows are handled separately via AO.
153 */
154vec3 skyDirect(Hit h, MaterialProps mat, vec3 Vdir)
155{
156 if (uSkyEnabled == 0) return vec3(0.0);
157
158 vec3 N = normalize(h.n);
159 vec3 U = normalize(uSkyUpDir);
160
161 float ndl = max(dot(N, U), 0.0);
162 if (ndl <= 0.0) return vec3(0.0);
163
164 // Simple cosine-weighted dome around U
165 vec3 Li = uSkyColor * uSkyIntensity;
166
167 // Diffuse only, no specular
168 return mat.albedo * (ndl / uPI) * Li;
169}

171// ---------------------------------------------------------------------------
172// Point light (shadowed, inverse-square falloff)
173// ---------------------------------------------------------------------------
174
175/**
176 * @brief Point light contribution with inverse-square falloff and shadows.
177 *
178 * For the analytic scene, the emissive point-light marker sphere is excluded
179 * from shadow tests via traceAnalyticIgnorePointLight().
180 */
181vec3 pointDirect(Hit h, MaterialProps mat, vec3 Vdir)
182{
183 if (uPointLightEnabled == 0) return vec3(0.0);
184
185 vec3 N = normalize(h.n);
186 vec3 V = normalize(Vdir);
187 vec3 toL = uPointLightPos - h.p;
188 float dist2 = dot(toL, toL);
189 if (dist2 <= 1e-6) return vec3(0.0);
190
191 float dist = sqrt(dist2);
192 vec3 L = toL / dist;
193 float ndl = max(dot(N, L), 0.0);
194 if (ndl <= 0.0) return vec3(0.0);
195
196 float eps = epsForDist(dist);
197 vec3 origin = h.p + L * eps;
198
199 bool blocked;
200 if (uUseBVH == 1) {
201 blocked = traceBVHShadow(origin, L, dist - eps);
202 } else {
203 Hit tmp;
204 // IMPORTANT: do NOT let the marker sphere shadow its own light
205 blocked = traceAnalyticIgnorePointLight(origin, L, tmp) && tmp.t < dist - eps;
206 }
207 if (blocked) return vec3(0.0);
208
209 // Inverse-square falloff
210 vec3 Li = uPointLightColor * (uPointLightIntensity / max(dist2, 1e-4));
211
212 float specStrength = (mat.type == 0) ? mat.specStrength : 0.0;
213 return shadeLambertPhong(N, V, L, Li, mat.albedo, specStrength, mat.gloss);
214}

220/**
221 * @brief Builds an orthonormal basis (T, B, N) from a normal.
222 *
223 * @param N Normalized surface normal.
224 * @param T Output tangent vector.
225 * @param B Output bitangent vector.
226 */
227void buildONB(in vec3 N, out vec3 T, out vec3 B) {
228 vec3 up = (abs(N.y) < 0.99) ? vec3(0.0, 1.0, 0.0) : vec3(1.0, 0.0, 0.0);
229 T = normalize(cross(up, N));
230 B = cross(N, T);
231}
232
233/**
234 * @brief Builds a local frame (t,b) around the disk light normal kLightN.
235 *
236 * Used to map 2D disk samples into world space for the soft area light.
237 */
238void buildLightFrame(out vec3 t, out vec3 b) {
239 t = normalize(abs(kLightN.y) < 0.99 ? cross(kLightN, vec3(0, 1, 0))
240 : cross(kLightN, vec3(1, 0, 0)));
241 b = cross(kLightN, t);
242}
243
244/**
245 * @brief Cosine-weighted hemisphere sample around normal N.
246 *
247 * @param N Surface normal (world space).
248 * @param u 2D random sample in [0,1]^2.
249 * @return Sampled direction on the hemisphere around N.
250 */
251vec3 sampleHemisphereCosine(vec3 N, vec2 u) {
252 // cosine-weighted sample in local space (around +Y)
253 float phi = 2.0 * uPI * u.x;
254 float r = sqrt(u.y);
255 float x = r * cos(phi);
256 float z = r * sin(phi);
257 float y = sqrt(max(0.0, 1.0 - u.y));
258 vec3 l = vec3(x, y, z); // local (+Y hemisphere)
259
260 // build tangent frame once
261 vec3 T, B;
262 buildONB(normalize(N), T, B);
263
264 // Map local sample into world space oriented around N
265 return normalize(l.x * T + l.z * B + l.y * N);
266}

272// ---- Direct lighting (analytic & BVH) with per-pixel CP rotation + frame LD rotation
273
274/**
275 * @brief Computes a per-pixel offset for disk-light sampling.
276 *
277 * Combines hash-based noise and a low-discrepancy sequence to rotate the
278 * sample pattern over time and across pixels, reducing structured noise.
279 */
280vec2 cpOffset(vec2 pix, int frame) {
281 // Per-pixel hash → [0,1)^2
282 vec2 h = vec2(
283 rand(pix, frame * 911),
284 rand(pix.yx, frame * 577)
285 );
286 // Per-frame low-discrepancy rotation
287 vec2 ld = ld2(frame);
288 return fract(h + ld);
289}

291/**
292 * @brief Standalone Phong specular evaluation helper.
293 *
294 * Used by some direct-light routines when a separate specular term is needed.
295 */
296vec3 evalPhongSpec(vec3 N, vec3 V, vec3 L, float gloss, float specStrength) {
297 if (specStrength <= 0.0) return vec3(0.0);
298 vec3 H = normalize(L + V);
299 float ndh = max(dot(N, H), 0.0);
300 float phong = pow(ndh, gloss);
301 return specStrength * phong * vec3(1.0);
302}

304// ---- Direct lighting (analytic) with cheap specular + hybrid lights
305
306/**
307 * @brief Direct lighting for the analytic scene (plane + spheres).
308 *
309 * Evaluates:
310 * - Disk area light with soft shadows.
311 * - Sun, sky, and point lights (hybrid analytic lights).
312 *
313 * @param h Primary hit.
314 * @param frame Current frame index.
315 * @param Vdir Direction from hit → camera.
316 * @return Total direct lighting contribution at h.
317 */
318vec3 directLight(Hit h, int frame, vec3 Vdir) {
319 vec3 N = normalize(h.n);
320 vec3 sum = vec3(0.0);
321
322 MaterialProps mat = getMaterial(h.mat);
323
324 // Orthonormal basis for disk light
325 vec3 t = normalize(abs(kLightN.y) < 0.99 ? cross(kLightN, vec3(0, 1, 0))
326 : cross(kLightN, vec3(1, 0, 0)));
327 vec3 b = cross(kLightN, t);
328
329 // Per-pixel rotated sampling
330 vec2 rot = cpOffset(gl_FragCoord.xy, uFrameIndex);
331
332 vec3 V = normalize(Vdir); // from hit → camera
333
334 // Existing soft disk area light
335 for (int i = 0; i < SOFT_SHADOW_SAMPLES; ++i) {
336 vec2 u = vec2(
337 rand(gl_FragCoord.xy + float(i), frame),
338 rand(gl_FragCoord.yx + float(31 * i + 7), frame)
339 );
340 u = fract(u + rot);
341
342 vec2 d = concentricSample(u) * kLightRadius;
343 vec3 xL = kLightCenter + t * d.x + b * d.y;
344
345 vec3 L = normalize(xL - h.p);
346 float ndl = max(dot(N, L), 0.0);
347 float cosThetaL = max(dot(-kLightN, L), 0.0);
348 float r2 = max(dot(xL - h.p, xL - h.p), 1e-4);
349
350 float geom = (ndl * cosThetaL) / r2;
351 float vis = occludedToward(h.p, xL) ? 0.0 : 1.0;
352
353 vec3 Li = kLightCol * geom * vis;
354
355 float specStrength = (mat.type == 0) ? mat.specStrength : 0.0;
356 sum += shadeLambertPhong(N, V, L, Li, mat.albedo, specStrength, mat.gloss);
357 }
358
359 sum /= float(SOFT_SHADOW_SAMPLES);
360
361 // Add hybrid lights
362 sum += sunDirect(h, mat, V);
363 sum += skyDirect(h, mat, V);
364 sum += pointDirect(h, mat, V);
365
366 return sum;
367}

369// ---- Direct lighting for BVH triangles (simple white plastic) + hybrid lights
370
371/**
372 * @brief Direct lighting for BVH triangle geometry.
373 *
374 * Uses a hard-coded "white plastic" material for the triangle mesh and
375 * reuses the same disk, sun, sky, and point lights as the analytic scene.
376 */
377vec3 directLightBVH(Hit h, int frame, vec3 Vdir) {
378 vec3 N = normalize(h.n);
379 vec3 sum = vec3(0.0);
380
381 // Hardcoded BVH material: white plastic
382 const vec3 albedo = vec3(0.85);
383 const float specStrength = 0.25;
384 const float gloss = 32.0;
385
386 vec3 t = normalize(abs(kLightN.y) < 0.99 ? cross(kLightN, vec3(0, 1, 0))
387 : cross(kLightN, vec3(1, 0, 0)));
388 vec3 b = cross(kLightN, t);
389
390 vec2 rot = cpOffset(gl_FragCoord.xy, uFrameIndex);
391 vec3 V = normalize(Vdir);
392
393 // Disk area light
394 for (int i = 0; i < SOFT_SHADOW_SAMPLES; ++i) {
395 vec2 u = vec2(
396 rand(gl_FragCoord.xy + float(i), frame),
397 rand(gl_FragCoord.yx + float(31 * i + 7), frame)
398 );
399 u = fract(u + rot);
400
401 vec2 d = concentricSample(u) * kLightRadius;
402 vec3 xL = kLightCenter + t * d.x + b * d.y;
403
404 vec3 L = normalize(xL - h.p);
405 float ndl = max(dot(N, L), 0.0);
406 float cosThetaL = max(dot(-kLightN, L), 0.0);
407 float r2 = max(dot(xL - h.p, xL - h.p), 1e-4);
408
409 float geom = (ndl * cosThetaL) / r2;
410 float vis = occludedToward(h.p, xL) ? 0.0 : 1.0;
411
412 vec3 Li = kLightCol * geom * vis;
413
414 sum += shadeLambertPhong(N, V, L, Li, albedo, specStrength, gloss);
415 }
416
417 sum /= float(SOFT_SHADOW_SAMPLES);
418
419 // Approximate analytic MaterialProps for hybrid lights
420 MaterialProps fakeMat;
421 fakeMat.albedo = albedo;
422 fakeMat.specStrength = specStrength;
423 fakeMat.gloss = gloss;
424 fakeMat.type = 0; // diffuse-ish
425 fakeMat.ior = 1.0;
426
427 sum += sunDirect(h, fakeMat, V);
428 sum += skyDirect(h, fakeMat, V);
429 sum += pointDirect(h, fakeMat, V);
430
431 return sum;
432}

435// ============================================================================
436// One-bounce diffuse GI (indirect lighting)
437// ============================================================================
438
439/**
440 * @brief One-bounce diffuse GI for the analytic scene.
441 *
442 * Shoots a single cosine-weighted bounce from the primary hit, then:
443 * - if it hits: computes direct lighting at the secondary point
444 * - if it misses: samples the sky
445 */
446vec3 oneBounceGIAnalytic(Hit h0, int frame, int seed) {
447 MaterialProps mat0 = getMaterial(h0.mat);
448 vec3 albedo0 = mat0.albedo;
449
450 vec3 N0 = normalize(h0.n);
451
452 // Random sample on hemisphere (per-pixel, per-seed)
453 vec2 u = vec2(
454 rand(gl_FragCoord.xy + float(seed * 13), frame),
455 rand(gl_FragCoord.yx + float(seed * 37), frame)
456 );
457
458 vec3 wi = sampleHemisphereCosine(N0, u);
459 float cosTheta = max(dot(N0, wi), 0.0);
460 if (cosTheta <= 0.0) return vec3(0.0);
461
462 // Trace from the hit using a normal-based offset (more robust than along wi)
463 vec3 origin = h0.p + N0 * uEPS;
464
465 Hit h1;
466 bool hit1 = traceAnalytic(origin, wi, h1);
467
468 vec3 Li;
469 if (hit1) {
470 // Direct lighting at the secondary point (includes all lights)
471 vec3 V1 = -wi;
472 Li = directLight(h1, frame, V1);
473 } else {
474 // Bounce to sky
475 Li = sky(wi);
476 }
477
478 // Lambertian throughput: albedo0 * (cosTheta / uPI)
479 return albedo0 * (cosTheta / uPI) * Li;
480}
481
482/**
483 * @brief One-bounce diffuse GI for BVH geometry with firefly clamping.
484 *
485 * Uses a cosine-weighted bounce, reuses directLightBVH at the secondary hit,
486 * and applies a luminance clamp to reduce extreme GI spikes.
487 */
488vec3 oneBounceGIBVH(Hit h0, int frame, int seed) {
489 // Hard-coded BVH albedo (same spirit as directLightBVH)
490 const vec3 albedo0 = vec3(0.85);
491 const float MAX_GI_LUM = 8.0; // tweak: 4–12 depending on light power
492 const float MIN_COS_THETA = 0.1; // avoid super-grazing bounces
493
494 // Random sample on hemisphere (per-pixel, per-seed)
495 vec2 u = vec2(
496 rand(gl_FragCoord.xy + float(seed * 19), frame),
497 rand(gl_FragCoord.yx + float(seed * 41), frame)
498 );
499
500 vec3 N0 = normalize(h0.n);
501 vec3 wi = sampleHemisphereCosine(N0, u); // cosine-weighted around N
502 float cosTheta = max(dot(N0, wi), 0.0);
503
504 // Discard very grazing bounces (they cause huge variance on tiny triangles)
505 if (cosTheta <= MIN_COS_THETA)
506 return vec3(0.0);
507
508 // IMPORTANT: offset along the surface normal, not along wi
509 vec3 origin = h0.p + N0 * uEPS;
510
511 Hit h1;
512 bool hit1 = traceBVH(origin, wi, h1);
513
514 vec3 Li;
515 if (hit1) {
516 vec3 V1 = -wi;
517 // Includes disk, sky directional, and point light
518 Li = directLightBVH(h1, frame, V1);
519 } else {
520 Li = sky(wi);
521 }
522
523 // Raw Lambertian contribution
524 vec3 contrib = albedo0 * (cosTheta / uPI) * Li;
525
526 // Luminance-based clamp to kill fireflies
527 float lum = dot(contrib, vec3(0.299, 0.587, 0.114));
528 if (lum > MAX_GI_LUM) {
529 float s = MAX_GI_LUM / max(lum, 1e-6);
530 contrib *= s;
531 }
532
533 return contrib;
534}

536// ============================================================================
537// Glass shading – soft thin refraction with local reflections
538// ============================================================================
539// wo = direction from hit -> camera (i.e. -rayDir), frame = random seed
540
541/**
542 * @brief Shading for glass materials in the analytic scene.
543 *
544 * Approximates:
545 * - Thin refraction (straight-through + softened bent refraction).
546 * - Local and environment reflections.
547 * - Fresnel blending between reflection and refraction using Schlick's approx.
548 */
549vec3 shadeGlass(const Hit h, const vec3 wo, const MaterialProps mat, int frame) {
550 vec3 N = normalize(h.n);
551 vec3 V = normalize(wo); // hit -> camera
552 vec3 I = -V; // camera -> hit
553
554 float ior = mat.ior;
555 float eta = 1.0 / max(ior, 1.0001); // air -> glass
556
557 // 0.0 = no distortion (just straight-through)
558 // 1.0 = fully physical refraction
559 const float distortionStrength = 0.45;
560
561 // ------------------------------------------------------------------------
562 // REFLECTION: env + local scene
563 // ------------------------------------------------------------------------
564 vec3 R = reflect(I, N);
565
566 // Env reflection (cubemap/sky)
567 vec3 reflectEnv = sky(R);
568
569 // Local reflection: what the sphere "sees" around it
570 vec3 reflectLocal = reflectEnv;
571 {
572 Hit hRefl;
573 if (traceAnalyticIgnoreGlass(h.p + R * uEPS, R, hRefl)) {
574 vec3 V2 = normalize(uCamPos - hRefl.p); // hit -> camera
575 reflectLocal = directLight(hRefl, frame, V2);
576 }
577 }
578
579 // Blend env + local so nearby objects show up as small highlights,
580 // but the sphere doesn't turn into a full-on mirror.
581 const float localReflWeight = 0.4; // tweak 0.2–0.6
582 vec3 reflectCol = mix(reflectEnv, reflectLocal, localReflWeight);
583
584 // ------------------------------------------------------------------------
585 // REFRACTION: straight-through + softened bent refraction
586 // ------------------------------------------------------------------------
587
588 // Straight-through color (no bending – acts like thin glass)
589 vec3 straightCol = vec3(0.0);
590 {
591 Hit hStraight;
592 if (traceAnalyticIgnoreGlass(h.p + I * uEPS, I, hStraight)) {
593 vec3 V2 = normalize(uCamPos - hStraight.p); // from hit -> camera
594 straightCol = directLight(hStraight, frame, V2);
595 } else {
596 straightCol = sky(I);
597 }
598 }
599
600 // Bent refraction color (physical direction, blended down)
601 float cosTheta = clamp(dot(-I, N), 0.0, 1.0);
602 float k = 1.0 - eta * eta * (1.0 - cosTheta * cosTheta);
603
604 vec3 refrCol = straightCol; // fallback: at worst, just see straight through
605
606 if (distortionStrength > 0.0 && k > 0.0) {
607 // Physical refraction direction
608 vec3 T_phys = normalize(refract(I, N, eta));
609
610 // Soften the effect to avoid crazy magnification
611 vec3 T = normalize(mix(I, T_phys, distortionStrength));
612
613 Hit hRefr;
614 vec3 bentCol;
615 if (traceAnalyticIgnoreGlass(h.p + T * uEPS, T, hRefr)) {
616 vec3 V2 = normalize(uCamPos - hRefr.p);
617 bentCol = directLight(hRefr, frame, V2);
618 } else {
619 bentCol = sky(T);
620 }
621
622 refrCol = mix(straightCol, bentCol, distortionStrength);
623 }
624
625 // Tint by glass albedo
626 refrCol *= mat.albedo;
627
628 // ------------------------------------------------------------------------
629 // Fresnel (Schlick) – mix reflection vs refraction
630 // ------------------------------------------------------------------------
631 float F0 = pow((ior - 1.0) / (ior + 1.0), 2.0);
632 float fresnel = F0 + (1.0 - F0) * pow(1.0 - cosTheta, 5.0);
633
634 // Center = mostly refraction, rim = more reflection
635 return mix(refrCol, reflectCol, fresnel);
636}

638// ============================================================================
639// Mirror shading helper – analytic scene only
640// ============================================================================
641
642/**
643 * @brief Perfect mirror shading for analytic scene materials.
644 *
645 * Casts a reflection ray into the analytic scene, optionally adds one-bounce
646 * GI at the reflected hit, and falls back to the environment if nothing is hit.
647 */
648vec3 shadeMirror(const Hit h, const vec3 wo, const MaterialProps mat, int frame) {
649 vec3 N = normalize(h.n);
650 vec3 I = -normalize(wo); // direction from hit → camera
651 vec3 R = reflect(I, N); // perfect mirror reflection
652 vec3 org = h.p + R * uEPS;
653
654 Hit h2;
655 bool hit2 = traceAnalytic(org, R, h2);
656
657 vec3 col;
658 if (hit2) {
659 // Direct lighting at the reflected hit (all lights)
660 vec3 V2 = -R;
661 col = directLight(h2, frame, V2);
662
663 // Optional one-bounce GI for the reflected point
664 if (uEnableGI == 1) {
665 int giSeed = frame * 131 + 17;
666 col += uGiScaleAnalytic * oneBounceGIAnalytic(h2, frame, giSeed);
667 }
668 } else {
669 // Fallback: environment or sky
670 if (uUseEnvMap == 1) {
671 col = texture(uEnvMap, R).rgb * uEnvIntensity;
672 } else {
673 col = sky(R);
674 }
675 }
676
677 // Apply mirror tint
678 col *= mat.albedo;
679
680 return col;
681}

687/**
688 * @brief Computes ambient occlusion factor around a hit point.
689 *
690 * Shoots several hemisphere rays around the normal and counts the fraction
691 * of rays that quickly hit geometry within a radius. The final AO factor
692 * is clamped and remapped to avoid fully black regions.
693 */
694float computeAO(Hit h, int frame) {
695 vec3 N = normalize(h.n);
696 int occludedCount = 0;
697
698 for (int i = 0; i < uAO_SAMPLES; ++i) {
699 // deterministic but decorrelated per-pixel/per-sample noise
700 vec2 u = vec2(
701 rand(gl_FragCoord.xy + float(37 * i + 3), frame),
702 rand(gl_FragCoord.yx + float(19 * i + 11), frame)
703 );
704
705 // cosine-weighted world-space direction around N
706 vec3 dir = sampleHemisphereCosine(N, u);
707
708 // ray origin slightly above the surface (offset along normal for robustness)
709 vec3 org = h.p + N * uAO_BIAS;
710
711 Hit tmp;
712 bool hitAny =
713 (uUseBVH == 1)
714 ? traceBVH(org, dir, tmp)
715 : traceAnalytic(org, dir, tmp);
716
717 // count as occluded only if something is reasonably close
718 if (hitAny && tmp.t < uAO_RADIUS) {
719 occludedCount++;
720 }
721 }
722
723 float occ = float(occludedCount) / float(uAO_SAMPLES); // 0..1
724 float ao = 1.0 - occ; // 1=open, 0=closed
725
726 // Keep a minimum so nothing goes pitch-black
727 ao = clamp(mix(uAO_MIN, 1.0, ao), uAO_MIN, 1.0);
728
729 return ao;
730}
