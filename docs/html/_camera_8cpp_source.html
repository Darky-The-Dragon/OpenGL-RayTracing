53// Recompute Front/Right/Up based on updated yaw/pitch angles.
54void Camera::UpdateCameraVectors() {
55 glm::vec3 front;
56 front.x = cos(glm::radians(Yaw)) * cos(glm::radians(Pitch));
57 front.y = sin(glm::radians(Pitch));
58 front.z = sin(glm::radians(Yaw)) * cos(glm::radians(Pitch));
59 Front = glm::normalize(front);
60
61 Right = glm::normalize(glm::cross(Front, WorldUp));
62 Up = glm::normalize(glm::cross(Right, Front));
63}
